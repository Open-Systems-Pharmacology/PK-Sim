CREATE TABLE tab_disease_states
(
    disease_state  text not null
        constraint tab_disease_states_pk
            primary key,
    display_name   text not null,
    description    text not null,
    implementation text
);

CREATE UNIQUE INDEX tab_disease_states_description_uindex
    on tab_disease_states (description);

    on tab_disease_states (disease_state);
CREATE UNIQUE INDEX tab_disease_states_display_name_uindex
    on tab_disease_states (display_name);
CREATE TABLE tab_population_disease_states
(
    population    text not null
        constraint tab_population_disease_states_tab_populations_population_fk
            references tab_populations
            on update cascade,
    disease_state text not null
        constraint tab_population_disease_states_tab_disease_states_disease_state_fk
            references tab_disease_states
            on update cascade,
    constraint tab_population_disease_states_pk
        primary key (population, disease_state)
);
